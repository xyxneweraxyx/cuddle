#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "include/dataframe.h"

static int g_passed = 0;
static int g_failed = 0;
static int g_total = 0;

/* ─── helpers ─────────────────────────────────────────────────── */

static void pass(const char *name)
{
    printf("  \033[32m[PASS]\033[0m %s\n", name);
    g_passed++;
    g_total++;
}

static void fail(const char *name, const char *reason)
{
    printf("  \033[31m[FAIL]\033[0m %s\n         → %s\n", name, reason);
    g_failed++;
    g_total++;
}

static void section(const char *title)
{
    printf("\n\033[1;34m══ %s ══\033[0m\n", title);
}

static void check_shape(const char *name, dataframe_t *df,
    int exp_rows, int exp_cols)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->nb_rows == exp_rows && df->nb_columns == exp_cols) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "shape (%d, %d) attendu (%d, %d)",
            df->nb_rows, df->nb_columns, exp_rows, exp_cols);
        fail(name, buf);
    }
    df_free(df);
}

static void check_null(const char *name, dataframe_t *df)
{
    if (!df) { pass(name); return; }
    fail(name, "attendu NULL, got non-NULL");
    df_free(df);
}

static void check_str_val(const char *name, dataframe_t *df,
    int col, int row, const char *expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    char *got = df->columns[col].data[row].str;
    if (got && strcmp(got, expected) == 0) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d] = \"%s\", attendu \"%s\"",
            col, row, got ? got : "NULL", expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_uint_val(const char *name, dataframe_t *df,
    int col, int row, unsigned int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    unsigned int got = df->columns[col].data[row].ui;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].ui = %u, attendu %u",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_int_val(const char *name, dataframe_t *df,
    int col, int row, int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    int got = df->columns[col].data[row].i;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].i = %d, attendu %d",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_float_val(const char *name, dataframe_t *df,
    int col, int row, float expected, float eps)
{
    char buf[256];
    float diff;

    if (!df) { fail(name, "df est NULL"); return; }
    float got = df->columns[col].data[row].f;
    diff = got - expected;
    if (diff < 0) diff = -diff;
    if (diff < eps) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].f = %f, attendu %f",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_bool_val(const char *name, dataframe_t *df,
    int col, int row, int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    int got = (int)df->columns[col].data[row].b;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].b = %d, attendu %d",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void test_head_independence(void)
{
    dataframe_t *orig;
    dataframe_t *head;

    section("df_head - INDEPENDANCE");

    orig = df_read_csv("test_csv/basic.csv", ",");
    head = df_head(orig, 2);
    if (!head) { fail("independance: head est NULL", ""); df_free(orig); return; }

    /* modifier l'original ne doit pas affecter head */
    orig->columns[0].data[0].str[0] = 'X';
    if (strcmp(head->columns[0].data[0].str, "Alice") == 0)
        pass("independance: deep copy, modif original sans effet");
    else
        fail("independance: shallow copy detectee", "Alice attendu");
    df_free(orig);
    df_free(head);

    /* sep independant */
    orig = df_read_csv("test_csv/semicolon.csv", ";");
    head = df_head(orig, 1);
    if (head && strcmp(head->sep, ";") == 0)
        pass("head: sep copie correctement");
    else
        fail("head: sep non copie", ";  attendu");
    df_free(orig);
    if (head) df_free(head);
}

{

static void test_get_value(void)
{
    dataframe_t *df;
    void *val;
    char buf[256];

    section("df_get_value");

    df = df_read_csv("test_csv/basic.csv", ",");
    if (!df) { fail("get_value: df NULL", ""); return; }

    val = df_get_value(df, 0, "name");
    if (val && strcmp(((data_t *)val)->str, "Alice") == 0)
        pass("get_value(0, name) = Alice");
    else
        fail("get_value(0, name)", "attendu Alice");

    val = df_get_value(df, 2, "name");
    if (val && strcmp(((data_t *)val)->str, "Charlie") == 0)
        pass("get_value(2, name) = Charlie");
    else
        fail("get_value(2, name)", "attendu Charlie");

    val = df_get_value(df, 0, "age");
    if (val && ((data_t *)val)->ui == 25)
        pass("get_value(0, age) = 25");
    else {
        snprintf(buf, sizeof(buf), "attendu 25, got %u",
            val ? ((data_t *)val)->ui : 0);
        fail("get_value(0, age)", buf);
    }

    val = df_get_value(df, 1, "age");
    if (val && ((data_t *)val)->ui == 30)
        pass("get_value(1, age) = 30");
    else
        fail("get_value(1, age)", "attendu 30");

    val = df_get_value(df, 0, "colonne_inexistante");
    if (!val)
        pass("get_value colonne inexistante -> NULL");
    else
        fail("get_value colonne inexistante", "attendu NULL");

    val = df_get_value(df, 99, "name");
    if (!val)
        pass("get_value row hors bornes -> NULL");
    else
        fail("get_value row hors bornes", "attendu NULL");

    val = df_get_value(df, 0, NULL);
    if (!val)
        pass("get_value column NULL -> NULL");
    else
        fail("get_value column NULL", "attendu NULL");

    val = df_get_value(NULL, 0, "name");
    if (!val)
        pass("get_value df NULL -> NULL");
    else
        fail("get_value df NULL", "attendu NULL");

    df_free(df);

    df = df_read_csv("test_csv/types_float.csv", ",");
    if (!df) { fail("get_value float: df NULL", ""); return; }
    val = df_get_value(df, 1, "price");
    if (val) {
        float diff = ((data_t *)val)->f - 3.14f;
        if (diff < 0) diff = -diff;
        if (diff < 0.01f)
            pass("get_value float: price[1] = 3.14");
        else
            fail("get_value float", "attendu 3.14");
    } else {
        fail("get_value float", "val est NULL");
    }

    df = df_read_csv("test_csv/types_bool.csv", ",");
    if (!df) { fail("get_value bool: df NULL", ""); return; }
    val = df_get_value(df, 0, "active");
    if (val && ((data_t *)val)->b == true)
        pass("get_value bool: active[0] = true");
    else
        fail("get_value bool", "attendu true");
    df_free(df);

{

    section("df_get_unique_values");

    pass("unique name: non-NULL");

    for (int i = 0; i < df->nb_rows; i++) {
        if (vals[i])
            count++;
    }
    if (count == 3) {
        pass("unique name: 3 valeurs uniques");
    } else {
        char buf[64];
        snprintf(buf, sizeof(buf), "got %d, attendu 3", count);
        fail("unique name: compte", buf);
    }
    free(vals);

    /* score: 10, 20, 30 (3 uniques sur 6) */
    vals = df_get_unique_values(df, "score");
    if (vals) {
        count = 0;
        for (int i = 0; i < df->nb_rows; i++) {
            if (vals[i]) count++;
        }
        if (count == 3)
            pass("unique score: 3 valeurs uniques");
        else {
            char buf[64];
            snprintf(buf, sizeof(buf), "got %d, attendu 3", count);
            fail("unique score", buf);
        }
        free(vals);
    } else {
        fail("unique score: NULL", "");
    }

    vals = df_get_unique_values(df, "colonne_inexistante");

        pass("unique colonne inexistante -> NULL");
    else {
        fail("unique colonne inexistante", "attendu NULL");

    }

    vals = df_get_unique_values(NULL, "name");
    if (!vals)
        pass("unique df NULL -> NULL");
    else {
        fail("unique df NULL", "attendu NULL");
        free(vals);
    }

    df_free(df);

    /* colonne avec que des valeurs identiques */
    df = df_read_csv("test_csv/zero_values.csv", ",");
    if (!df) { fail("unique zeros: df NULL", ""); return; }
    vals = df_get_unique_values(df, "value");

        count = 0;
        for (int i = 0; i < df->nb_rows; i++) {
            if (vals[i]) count++;

            pass("unique zeros: 1 valeur unique (0)");

            fail("unique zeros", buf);
        }
        free(vals);
    } else {
        fail("unique zeros: NULL", "");
    }
    df_free(df);
}

static void *double_int(void *val)
{
    int *n = (int *)val;
    int *result = malloc(sizeof(int));

    *result = *n * 2;
    return result;
}

static void *add_one_uint(void *val)
{
    unsigned int *n = (unsigned int *)val;
    unsigned int *result = malloc(sizeof(unsigned int));

    *result = *n + 1;
    return result;
}

static void *negate_float(void *val)
{
    float *f = (float *)val;
    float *result = malloc(sizeof(float));

    *result = -*f;
    return result;
}

static void *upper_first(void *val)
{
    char **str = (char **)val;
    char *result = malloc(strlen(*str) + 1);

    strcpy(result, *str);
    if (result[0] >= 'a' && result[0] <= 'z')
        result[0] -= 32;
    return &result;
}

static void test_apply(void)
{
    dataframe_t *df;
    dataframe_t *orig;
    dataframe_t *result;

    section("df_apply");

    /* INT */
    orig = df_read_csv("test_csv/apply_int.csv", ",");
    if (!orig) { fail("apply int: orig NULL", ""); return; }
    result = df_apply(orig, "value", double_int);
    if (!result) {
        fail("apply int: result NULL", "");
        df_free(orig);
        return;
    }
    check_int_val("apply double_int: col[1][0] = 20", result, 1, 0, 20);

    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, "value", double_int);
    check_int_val("apply double_int: col[1][1] = 40", result, 1, 1, 40);
    df_free(orig);

    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, "value", double_int);
    check_int_val("apply double_int: col[1][2] = 60", result, 1, 2, 60);
    df_free(orig);

    /* colonne non modifiee reste intacte */
    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, "value", double_int);
    check_uint_val("apply: col non ciblee intacte col[0][0] = 1", result, 0, 0, 1);
    df_free(orig);

    /* UINT */
    orig = df_read_csv("test_csv/types_uint.csv", ",");
    if (!orig) { fail("apply uint: orig NULL", ""); return; }
    result = df_apply(orig, "score", add_one_uint);
    check_uint_val("apply add_one_uint: col[1][0] = 101", result, 1, 0, 101);
    df_free(orig);

    orig = df_read_csv("test_csv/types_uint.csv", ",");
    result = df_apply(orig, "score", add_one_uint);
    check_uint_val("apply add_one_uint: col[1][2] = 51", result, 1, 2, 51);
    df_free(orig);

    /* FLOAT */
    orig = df_read_csv("test_csv/types_float.csv", ",");
    if (!orig) { fail("apply float: orig NULL", ""); return; }
    result = df_apply(orig, "price", negate_float);
    check_float_val("apply negate_float: col[1][0] = -9.99", result, 1, 0, -9.99f, 0.01f);
    df_free(orig);

    orig = df_read_csv("test_csv/types_float.csv", ",");
    result = df_apply(orig, "price", negate_float);
    check_float_val("apply negate_float: col[1][1] = -3.14", result, 1, 1, -3.14f, 0.01f);
    df_free(orig);

    /* erreurs */
    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, "colonne_inexistante", double_int);
    if (!result)
        pass("apply colonne inexistante -> NULL");
    else {
        fail("apply colonne inexistante", "attendu NULL");
        df_free(result);
    }
    df_free(orig);

    result = df_apply(NULL, "value", double_int);
    if (!result)
        pass("apply df NULL -> NULL");
    else {
        fail("apply df NULL", "attendu NULL");
        df_free(result);
    }

    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, NULL, double_int);
    if (!result)
        pass("apply column NULL -> NULL");
    else {
        fail("apply column NULL", "attendu NULL");
        df_free(result);
    }
    df_free(orig);

    /* independance original */
    orig = df_read_csv("test_csv/apply_int.csv", ",");
    result = df_apply(orig, "value", double_int);
    if (result && orig->columns[1].data[0].i == 10)
        pass("apply: original inchange apres apply");
    else
        fail("apply: original modifie", "attendu 10");
    if (result) df_free(result);
    df_free(orig);
}
