#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "include/dataframe.h"

static int g_passed = 0;
static int g_failed = 0;
static int g_total = 0;

/* ─── helpers ─────────────────────────────────────────────────── */

static void pass(const char *name)
{
    printf("  \033[32m[PASS]\033[0m %s\n", name);
    g_passed++;
    g_total++;
}

static void fail(const char *name, const char *reason)
{
    printf("  \033[31m[FAIL]\033[0m %s\n         → %s\n", name, reason);
    g_failed++;
    g_total++;
}

static void section(const char *title)
{
    printf("\n\033[1;34m══ %s ══\033[0m\n", title);
}

static void check_null(const char *name, dataframe_t *df)
{
    if (!df) { pass(name); return; }
    fail(name, "attendu NULL, got non-NULL");
    df_free(df);
}

static void check_shape(const char *name, dataframe_t *df,
    int exp_rows, int exp_cols)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->nb_rows == exp_rows && df->nb_columns == exp_cols) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "shape (%d, %d) attendu (%d, %d)",
            df->nb_rows, df->nb_columns, exp_rows, exp_cols);
        fail(name, buf);
    }
    df_free(df);
}

static void check_str_val(const char *name, dataframe_t *df,
    int col, int row, const char *expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    char *got = df->columns[col].data[row].str;
    if (got && strcmp(got, expected) == 0) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d] = \"%s\", attendu \"%s\"",
            col, row, got ? got : "NULL", expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_uint_val(const char *name, dataframe_t *df,
    int col, int row, unsigned int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    unsigned int got = df->columns[col].data[row].ui;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].ui = %u, attendu %u",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_int_val(const char *name, dataframe_t *df,
    int col, int row, int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    int got = df->columns[col].data[row].i;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].i = %d, attendu %d",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_float_val(const char *name, dataframe_t *df,
    int col, int row, float expected, float eps)
{
    char buf[256];
    float diff;

    if (!df) { fail(name, "df est NULL"); return; }
    float got = df->columns[col].data[row].f;
    diff = got - expected;
    if (diff < 0) diff = -diff;
    if (diff < eps) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d].f = %f, attendu %f",
            col, row, got, expected);
        fail(name, buf);
    }
    df_free(df);
}

static void check_col_type(const char *name, dataframe_t *df,
    int col, column_type_t expected)
{
    char buf[256];
    const char *types[] = {"BOOL","INT","UINT","FLOAT","STRING","UNDEFINED"};

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->columns[col].type == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d] = %s, attendu %s",
            col, types[df->columns[col].type], types[expected]);
        fail(name, buf);
    }
    df_free(df);
}

/* ─── filter callbacks ────────────────────────────────────────── */

static bool age_gt_28(void *val)
{
    return ((data_t *)val)->ui > 28;
}

static bool age_eq_25(void *val)
{
    return ((data_t *)val)->ui == 25;
}

static bool active_true(void *val)
{
    return ((data_t *)val)->b == true;
}

static bool score_gt_150(void *val)
{
    return ((data_t *)val)->ui > 150;
}

static bool never_true(void *val)
{
    (void)val;
    return false;
}

static bool always_true(void *val)
{
    (void)val;
    return true;
}

static bool ratio_gt_3(void *val)
{
    return ((data_t *)val)->f > 3.0f;
}

static bool val_negative(void *val)
{
    return ((data_t *)val)->i < 0;
}

/* ─── sort callbacks ──────────────────────────────────────────── */

static bool uint_asc(void *a, void *b)
{
    return ((data_t *)a)->ui <= ((data_t *)b)->ui;
}

static bool uint_desc(void *a, void *b)
{
    return ((data_t *)a)->ui >= ((data_t *)b)->ui;
}

static bool int_asc(void *a, void *b)
{
    return ((data_t *)a)->i <= ((data_t *)b)->i;
}

static bool float_asc(void *a, void *b)
{
    return ((data_t *)a)->f <= ((data_t *)b)->f;
}

    section("df_filter - INDEPENDANCE");

    filtered = df_filter(orig, "age", age_gt_28);
    if (!filtered) {
        fail("independance filter: NULL", "");
        df_free(orig);
        return;
    }
    orig->columns[0].data[1].str[0] = 'X';
    if (strcmp(filtered->columns[0].data[0].str, "Bob") == 0)

    orig = df_read_csv("test_csv/multi_types.csv", ",");

    if (!converted) {
        fail("to_type independance: NULL", "");
        df_free(orig);
        return;
    }
    if (orig->columns[1].type == UINT)
        pass("to_type independance: type original inchange");
    else
        fail("to_type independance", "type original modifie");
    if (orig->columns[1].data[0].ui == 25)
        pass("to_type independance: valeur originale inchangee");
    else
        fail("to_type independance", "valeur originale modifiee");
    df_free(orig);
    df_free(converted);
}
