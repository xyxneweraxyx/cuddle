#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "include/dataframe.h"

static int g_passed = 0;
static int g_failed = 0;
static int g_total = 0;

/* ─── helpers ─────────────────────────────────────────────────── */

static void pass(const char *name)
{
    printf("  \033[32m[PASS]\033[0m %s\n", name);
    g_passed++;
    g_total++;
}

static void fail(const char *name, const char *reason)
{
    printf("  \033[31m[FAIL]\033[0m %s\n         → %s\n", name, reason);
    g_failed++;
    g_total++;
}

static void section(const char *title)
{
    printf("\n\033[1;34m══ %s ══\033[0m\n", title);
}

static void check_null(const char *name, dataframe_t *df)
{
    if (!df) { pass(name); return; }
    fail(name, "attendu NULL, got non-NULL");
    df_free(df);
}

static void check_shape(const char *name, dataframe_t *df,
    int exp_rows, int exp_cols)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->nb_rows == exp_rows && df->nb_columns == exp_cols) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "shape (%d, %d) attendu (%d, %d)",
            df->nb_rows, df->nb_columns, exp_rows, exp_cols);
        fail(name, buf);
    }
}

static void check_uint(const char *name, dataframe_t *df,
    int col, int row, unsigned int expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    unsigned int got = df->columns[col].data[row].ui;
    if (got == expected) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d] = %u, attendu %u",
            col, row, got, expected);
        fail(name, buf);
    }
}

static void check_float(const char *name, dataframe_t *df,
    int col, int row, float expected, float eps)
{
    char buf[256];
    float diff;

    if (!df) { fail(name, "df est NULL"); return; }
    float got = df->columns[col].data[row].f;
    diff = got - expected;
    if (diff < 0) diff = -diff;
    if (diff < eps) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d] = %f, attendu %f",
            col, row, got, expected);
        fail(name, buf);
    }
}

static void check_str(const char *name, dataframe_t *df,
    int col, int row, const char *expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    char *got = df->columns[col].data[row].str;
    if (got && strcmp(got, expected) == 0) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d][%d] = \"%s\", attendu \"%s\"",
            col, row, got ? got : "NULL", expected);
        fail(name, buf);
    }
}

static void check_col_name(const char *name, dataframe_t *df,
    int col, const char *expected)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    if (strcmp(df->columns[col].name, expected) == 0) {
        pass(name);
    } else {
        snprintf(buf, sizeof(buf), "col[%d].name = \"%s\", attendu \"%s\"",
            col, df->columns[col].name, expected);
        fail(name, buf);
    }
}

/* ─── agg callbacks ───────────────────────────────────────────── */

static void *sum_uint(void **values, int nb)
{
    unsigned int *result = malloc(sizeof(unsigned int));

    *result = 0;
    for (int i = 0; i < nb; i++)
        *result += ((data_t *)values[i])->ui;
    return result;
}

static void *sum_float(void **values, int nb)
{
    float *result = malloc(sizeof(float));

    *result = 0.0f;
    for (int i = 0; i < nb; i++)
        *result += ((data_t *)values[i])->f;
    return result;
}

static void *mean_uint(void **values, int nb)
{
    float *result = malloc(sizeof(float));

    *result = 0.0f;
    for (int i = 0; i < nb; i++)
        *result += (float)((data_t *)values[i])->ui;
    *result /= (float)nb;
    return result;
}

static void *max_uint(void **values, int nb)
{
    unsigned int *result = malloc(sizeof(unsigned int));

    *result = ((data_t *)values[0])->ui;
    for (int i = 1; i < nb; i++)
        if (((data_t *)values[i])->ui > *result)
            *result = ((data_t *)values[i])->ui;
    return result;
}

static void *count_rows(void **values, int nb)
{
    unsigned int *result = malloc(sizeof(unsigned int));

    (void)values;
    *result = (unsigned int)nb;
    return result;
}

/* ═══════════════════════════════════════════════════════════════ */
/*  SHAPE ET COLONNES                                              */
/* ═══════════════════════════════════════════════════════════════ */

static void test_shape(void)
{
    dataframe_t *df;
    dataframe_t *orig;
    const char *to_agg[] = {"score", NULL};
    const char *two_agg[] = {"score", "age", NULL};

    section("df_groupby - SHAPE ET COLONNES");

    /* 2 groupes (Paris, London) x 1 colonne agrégée + groupeur = 2 cols */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", to_agg, sum_uint);
    check_shape("groupby city/score: shape (2, 2)", df, 2, 2);
    df_free(df);
    df_free(orig);

    /* 2 colonnes agrégées = 3 colonnes au total */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", two_agg, sum_uint);
    check_shape("groupby city/score+age: shape (2, 3)", df, 2, 3);
    df_free(df);
    df_free(orig);

    /* 3 groupes uniques */
    orig = df_read_csv("test_csv/multi_types.csv", ",");
    const char *agg2[] = {"score", NULL};
    df = df_groupby(orig, "active", agg2, sum_uint);
    check_shape("groupby active: shape (2, 2)", df, 2, 2);
    df_free(df);
    df_free(orig);

    /* noms des colonnes préservés */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", to_agg, sum_uint);
    if (df) {
        check_col_name("groupby: col[0].name = city", df, 0, "city");
        check_col_name("groupby: col[1].name = score", df, 1, "score");
    }
    df_free(df);
    df_free(orig);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  SOMME                                                          */

static void test_sum(void)
{
    dataframe_t *df;
    dataframe_t *orig;
    const char *to_agg[] = {"score", NULL};

    section("df_groupby - SOMME (UINT)");

    /*
    ** cities.csv:
    ** name,city,age,score
    ** Alice,Paris,25,100
    ** Bob,London,30,200
    ** Charlie,Paris,35,300
    ** Dave,London,40,400
    ** Paris: 100+300=400 / London: 200+400=600
    */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", to_agg, sum_uint);
    if (!df) { fail("sum: df NULL", ""); df_free(orig); return; }

    /* col[0] = groupeur city, col[1] = score agrégé */
    check_str("groupby sum: city[0] = Paris", df, 0, 0, "Paris");
    check_str("groupby sum: city[1] = London", df, 0, 1, "London");
    check_uint("groupby sum: score Paris = 400", df, 1, 0, 400);
    check_uint("groupby sum: score London = 600", df, 1, 1, 600);
    df_free(df);
    df_free(orig);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  INDÉPENDANCE                                                   */
/* ═══════════════════════════════════════════════════════════════ */

static void test_independence(void)
{
    dataframe_t *orig;
    dataframe_t *result;
    const char *to_agg[] = {"score", NULL};

    section("df_groupby - INDÉPENDANCE");

    orig = df_read_csv("test_csv/cities.csv", ",");
    result = df_groupby(orig, "city", to_agg, sum_uint);
    if (!result) {
        fail("independance: df NULL", "");
        df_free(orig);
        return;
    }
    orig->columns[2].data[0].ui = 9999;
    if (result->columns[1].data[0].ui == 400)
        pass("groupby independance: deep copy OK");
    else
        fail("groupby independance", "shallow copy detectee");
    df_free(orig);
    df_free(result);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  ERREURS                                                        */
/* ═══════════════════════════════════════════════════════════════ */

static void test_errors(void)
{
    dataframe_t *df;
    dataframe_t *orig;
    const char *to_agg[] = {"score", NULL};
    const char *bad_agg[] = {"colonne_inexistante", NULL};

    section("df_groupby - ERREURS");

    df = df_groupby(NULL, "city", to_agg, sum_uint);
    check_null("groupby df NULL -> NULL", df);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, NULL, to_agg, sum_uint);
    check_null("groupby aggregate_by NULL -> NULL", df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", NULL, sum_uint);
    check_null("groupby to_aggregate NULL -> NULL", df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", to_agg, NULL);
    check_null("groupby agg_func NULL -> NULL", df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "colonne_inexistante", to_agg, sum_uint);
    check_null("groupby aggregate_by inexistant -> NULL", df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_groupby(orig, "city", bad_agg, sum_uint);
    check_null("groupby to_aggregate inexistant -> NULL", df);
    df_free(orig);
}
