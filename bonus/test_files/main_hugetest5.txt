#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "include/dataframe.h"

static int g_passed = 0;
static int g_failed = 0;
static int g_total = 0;

/* ─── helpers ─────────────────────────────────────────────────── */

static void pass(const char *name)
{
    printf("  \033[32m[PASS]\033[0m %s\n", name);
    g_passed++;
    g_total++;
}

static void fail(const char *name, const char *reason)
{
    printf("  \033[31m[FAIL]\033[0m %s\n         → %s\n", name, reason);
    g_failed++;
    g_total++;
}

static void section(const char *title)
{
    printf("\n\033[1;33m══ %s ══\033[0m\n", title);
}

static void check_null(const char *name, void *ptr)
{
    if (!ptr) { pass(name); return; }
    fail(name, "attendu NULL");
}

static void check_not_null(const char *name, void *ptr)
{
    if (ptr) { pass(name); return; }
    fail(name, "attendu non-NULL, got NULL");
}

static void check_shape(const char *name, dataframe_t *df,
    int exp_rows, int exp_cols)
{
    char buf[256];

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->nb_rows == exp_rows && df->nb_columns == exp_cols) {
        pass(name);
        return;
    }
    snprintf(buf, sizeof(buf), "shape (%d,%d) attendu (%d,%d)",
        df->nb_rows, df->nb_columns, exp_rows, exp_cols);
    fail(name, buf);
}

static void check_str(const char *name, dataframe_t *df,
    int col, int row, const char *expected)
{
    char buf[256];
    char *got;

    if (!df) { fail(name, "df est NULL"); return; }
    got = df->columns[col].data[row].str;
    if (got && strcmp(got, expected) == 0) { pass(name); return; }
    snprintf(buf, sizeof(buf), "col[%d][%d]=\"%s\", attendu \"%s\"",
        col, row, got ? got : "NULL", expected);
    fail(name, buf);
}

static void check_uint(const char *name, dataframe_t *df,
    int col, int row, unsigned int expected)
{
    char buf[256];
    unsigned int got;

    if (!df) { fail(name, "df est NULL"); return; }
    got = df->columns[col].data[row].ui;
    if (got == expected) { pass(name); return; }
    snprintf(buf, sizeof(buf), "col[%d][%d]=%u, attendu %u",
        col, row, got, expected);
    fail(name, buf);
}

static void check_int(const char *name, dataframe_t *df,
    int col, int row, int expected)
{
    char buf[256];
    int got;

    if (!df) { fail(name, "df est NULL"); return; }
    got = df->columns[col].data[row].i;
    if (got == expected) { pass(name); return; }
    snprintf(buf, sizeof(buf), "col[%d][%d]=%d, attendu %d",
        col, row, got, expected);
    fail(name, buf);
}

static void check_float(const char *name, dataframe_t *df,
    int col, int row, float expected, float eps)
{
    char buf[256];
    float got;
    float diff;

    if (!df) { fail(name, "df est NULL"); return; }
    got = df->columns[col].data[row].f;
    diff = got - expected;
    if (diff < 0) diff = -diff;
    if (diff < eps) { pass(name); return; }
    snprintf(buf, sizeof(buf), "col[%d][%d]=%f, attendu %f",
        col, row, got, expected);
    fail(name, buf);
}

static void check_bool(const char *name, dataframe_t *df,
    int col, int row, bool expected)
{
    char buf[256];
    bool got;

    if (!df) { fail(name, "df est NULL"); return; }
    got = df->columns[col].data[row].b;
    if (got == expected) { pass(name); return; }
    snprintf(buf, sizeof(buf), "col[%d][%d]=%s, attendu %s",
        col, row, got ? "true" : "false",
        expected ? "true" : "false");
    fail(name, buf);
}

static void check_col_type(const char *name, dataframe_t *df,
    int col, column_type_t expected)
{
    char buf[256];
    const char *types[] = {"BOOL","INT","UINT","FLOAT","STRING","UNDEF"};

    if (!df) { fail(name, "df est NULL"); return; }
    if (df->columns[col].type == expected) { pass(name); return; }
    snprintf(buf, sizeof(buf), "type=%s, attendu %s",
        types[df->columns[col].type], types[expected]);
    fail(name, buf);
}

/* ─── CSV generator ───────────────────────────────────────────── */

static void gen_csv(void)
{
    FILE *f;

    system("mkdir -p test_csv");

    /* basic.csv: name(str), age(uint), city(str) */
    f = fopen("test_csv/basic.csv", "w");
    fprintf(f, "name,age,city\n");
    fprintf(f, "Alice,25,Paris\n");
    fprintf(f, "Bob,30,London\n");
    fprintf(f, "Charlie,35,Tokyo\n");
    fclose(f);

    /* numbers.csv: label(str), val_int(int), val_uint(uint), val_float(float) */
    f = fopen("test_csv/numbers.csv", "w");
    fprintf(f, "label,val_int,val_uint,val_float\n");
    fprintf(f, "a,-10,10,1.5\n");
    fprintf(f, "b,-20,20,2.5\n");
    fprintf(f, "c,-30,30,3.5\n");
    fclose(f);

    /* str_numbers.csv: colonnes STRING contenant des nombres */
    f = fopen("test_csv/str_numbers.csv", "w");
    fprintf(f, "name,age_str,score_str,ratio_str\n");
    fprintf(f, "Alice,25,100,1.5\n");
    fprintf(f, "Bob,30,200,2.5\n");
    fprintf(f, "Charlie,35,300,3.5\n");
    fclose(f);

    /* bools.csv: name(str), active(bool), score(uint) */
    f = fopen("test_csv/bools.csv", "w");
    fprintf(f, "name,active,score\n");
    fprintf(f, "Alice,true,100\n");
    fprintf(f, "Bob,false,200\n");
    fprintf(f, "Charlie,true,300\n");
    fclose(f);

    /* cities.csv: name(str), city(str), score(uint) */
    f = fopen("test_csv/cities.csv", "w");
    fprintf(f, "name,city,score\n");
    fprintf(f, "Alice,Paris,100\n");
    fprintf(f, "Bob,London,200\n");
    fprintf(f, "Charlie,Paris,300\n");
    fprintf(f, "Dave,London,400\n");
    fclose(f);
}

/*  HEAD / TAIL — NEGATIFS                                         */

static void test_head_tail_negative(void)
{
    dataframe_t *df;
    dataframe_t *orig;

    section("HEAD / TAIL — nb_rows NEGATIF");

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_head(orig, -1);
    check_null("head(-1) -> NULL", df);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_head(orig, -100);
    check_null("head(-100) -> NULL", df);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_tail(orig, -1);
    check_null("tail(-1) -> NULL", df);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_tail(orig, -100);
    check_null("tail(-100) -> NULL", df);
    df_free(df);
    df_free(orig);

    /* head(0) -> NULL ou df vide selon implementation */
orig = df_read_csv("test_csv/basic.csv", ",");
df = df_head(orig, 0);
check_not_null("head(0) -> non-NULL", df);
if (df) check_shape("head(0): shape (0,3)", df, 0, 3);
df_free(df);
df_free(orig);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  FILTER — STRING                                                */
/* ═══════════════════════════════════════════════════════════════ */

static bool city_is_paris(void *val)
{
    return strcmp((char *)val, "Paris") == 0;
}

static bool city_is_london(void *val)
{
    return strcmp((char *)val, "London") == 0;
}

static bool name_starts_a(void *val)
{
    return ((char *)val)[0] == 'A';
}

static bool age_gt_25(void *val)
{
    return ((data_t *)val)->ui > 25;
}

static void test_filter_string(void)
{
    dataframe_t *df;
    dataframe_t *orig;

    section("FILTER — COLONNE STRING");

    /* filter par city == Paris: Alice, Charlie */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_paris);
    check_not_null("filter city=Paris: non-NULL", df);
    check_shape("filter city=Paris: shape (2,3)", df, 2, 3);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_paris);
    check_str("filter city=Paris: name[0]=Alice", df, 0, 0, "Alice");
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_paris);
    check_str("filter city=Paris: name[1]=Charlie", df, 0, 1, "Charlie");
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_paris);
    check_str("filter city=Paris: city[0]=Paris", df, 1, 0, "Paris");
    df_free(df);
    df_free(orig);

    /* filter par city == London: Bob, Dave */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_london);
    check_shape("filter city=London: shape (2,3)", df, 2, 3);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_filter(orig, "city", city_is_london);
    check_str("filter city=London: name[0]=Bob", df, 0, 0, "Bob");
    df_free(df);
    df_free(orig);

    /* filter par name commence par A */
    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_filter(orig, "name", name_starts_a);
    check_shape("filter name[0]='A': shape (1,3)", df, 1, 3);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_filter(orig, "name", name_starts_a);
    check_str("filter name[0]='A': Alice", df, 0, 0, "Alice");
    df_free(df);
    df_free(orig);

    /* filter UINT fonctionne toujours */
    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_filter(orig, "age", age_gt_25);
    check_shape("filter age>25: shape (2,3)", df, 2, 3);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/basic.csv", ",");
    df = df_filter(orig, "age", age_gt_25);
    check_str("filter age>25: name[0]=Bob", df, 0, 0, "Bob");
    df_free(df);
    df_free(orig);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  SORT — STRING                                                  */
/* ═══════════════════════════════════════════════════════════════ */

static bool str_asc(void *a, void *b)
{
    return strcmp((char *)a, (char *)b) > 0;
}

static bool str_desc(void *a, void *b)
{
    return strcmp((char *)a, (char *)b) < 0;
}

static bool uint_asc(void *a, void *b)
{
    return ((data_t *)a)->ui > ((data_t *)b)->ui;
}

static void test_sort_string(void)
{
    dataframe_t *df;
    dataframe_t *orig;

    section("SORT — COLONNE STRING");

    /* tri asc par name: Alice < Bob < Charlie < Dave */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_asc);
    check_not_null("sort name asc: non-NULL", df);
    check_str("sort name asc: [0]=Alice", df, 0, 0, "Alice");
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_asc);
    check_str("sort name asc: [3]=Dave", df, 0, 3, "Dave");
    df_free(df);
    df_free(orig);

    /* tri desc par name: Dave > Charlie > Bob > Alice */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_desc);
    check_str("sort name desc: [0]=Dave", df, 0, 0, "Dave");
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_desc);
    check_str("sort name desc: [3]=Alice", df, 0, 3, "Alice");
    df_free(df);
    df_free(orig);

    /* coherence: score suit le tri par name */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_asc);
    check_uint("sort name asc: score[0]=100 (Alice)", df, 2, 0, 100);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "name", str_desc);
    check_uint("sort name desc: score[0]=400 (Dave)", df, 2, 0, 400);
    df_free(df);
    df_free(orig);

    /* tri par city: London London Paris Paris */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "city", str_asc);
    check_str("sort city asc: [0]=London", df, 1, 0, "London");
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "city", str_asc);
    check_str("sort city asc: [2]=Paris", df, 1, 2, "Paris");
    df_free(df);
    df_free(orig);

    /* UINT sort toujours fonctionnel */
    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "score", uint_asc);
    check_uint("sort score asc: [0]=100", df, 2, 0, 100);
    df_free(df);
    df_free(orig);

    orig = df_read_csv("test_csv/cities.csv", ",");
    df = df_sort(orig, "score", uint_asc);
    check_str("sort score asc: name[0]=Alice", df, 0, 0, "Alice");
    df_free(df);
    df_free(orig);
}

/* ═══════════════════════════════════════════════════════════════ */
/*  GET UNIQUE VALUES — STRING                                     */
/* ═══════════════════════════════════════════════════════════════ */

static void test_get_unique_string(void)
{
    dataframe_t *df;
    void **vals;
    int count;

    section("GET UNIQUE VALUES — STRING");

    df = df_read_csv("test_csv/cities.csv", ",");

    /* city: Paris, London (2 uniques sur 4) */
    vals = df_get_unique_values(df, "city");
    check_not_null("unique city: non-NULL", vals);
    if (vals) {
        count = 0;
        for (int i = 0; i < df->nb_rows; i++)
            if (vals[i]) count++;
        if (count == 2)
            pass("unique city: 2 valeurs uniques");
        else {
            char buf[64];
            snprintf(buf, sizeof(buf), "got %d, attendu 2", count);
            fail("unique city: compte", buf);
        }
        /* verifier que ce sont des char* lisibles */
        if (vals[0] && strcmp((char *)vals[0], "Paris") == 0)
            pass("unique city: vals[0] = Paris");
        else
            fail("unique city: vals[0]", "attendu Paris");
        if (vals[1] && strcmp((char *)vals[1], "London") == 0)
            pass("unique city: vals[1] = London");
        else
            fail("unique city: vals[1]", "attendu London");
        free(vals);
    }

    /* name: tous uniques (4 sur 4) */
    vals = df_get_unique_values(df, "name");
    if (vals) {
        count = 0;
        for (int i = 0; i < df->nb_rows; i++)
            if (vals[i]) count++;
        if (count == 4)
            pass("unique name: 4 valeurs uniques");
        else {
            char buf[64];
            snprintf(buf, sizeof(buf), "got %d, attendu 4", count);
            fail("unique name: compte", buf);
        }
        /* verifier lisibilite */
        if (vals[0] && strcmp((char *)vals[0], "Alice") == 0)
            pass("unique name: vals[0] = Alice");
        else
            fail("unique name: vals[0]", "attendu Alice");
        free(vals);
    }

    df_free(df);

    /* UINT unique */
    df = df_read_csv("test_csv/basic.csv", ",");
    vals = df_get_unique_values(df, "age");
    if (vals) {
        count = 0;
        for (int i = 0; i < df->nb_rows; i++)
            if (vals[i]) count++;
        if (count == 3)
            pass("unique age: 3 valeurs uniques");
        else {
            char buf[64];
            snprintf(buf, sizeof(buf), "got %d, attendu 3", count);
            fail("unique age: compte", buf);
        }
        if (((data_t *)vals[0])->ui == 25)
            pass("unique age: vals[0] = 25");
        else
            fail("unique age: vals[0]", "attendu 25");
        free(vals);
    }
    df_free(df);
}

static void test_to_type_str_to_num(void)
{
    dataframe_t *df;
    dataframe_t *tmp;
    dataframe_t *orig;

    section("TO TYPE — STRING -> NUMERIQUE");

    /* str -> uint: age UINT -> STRING -> UINT */
    orig = df_read_csv("test_csv/basic.csv", ",");
    tmp = df_to_type(orig, "age", STRING);
    df = df_to_type(tmp, "age", UINT);
    check_not_null("str->uint: non-NULL", df);
    check_col_type("str->uint: type=UINT", df, 1, UINT);
    check_uint("str->uint: Alice=25", df, 1, 0, 25);
    check_uint("str->uint: Bob=30", df, 1, 1, 30);
    check_uint("str->uint: Charlie=35", df, 1, 2, 35);
    df_free(df); df_free(tmp); df_free(orig);

    /* str -> int: age UINT -> STRING -> INT */
    orig = df_read_csv("test_csv/basic.csv", ",");
    tmp = df_to_type(orig, "age", STRING);
    df = df_to_type(tmp, "age", INT);
    check_not_null("str->int: non-NULL", df);
    check_col_type("str->int: type=INT", df, 1, INT);
    check_int("str->int: Alice=25", df, 1, 0, 25);
    check_int("str->int: Charlie=35", df, 1, 2, 35);
    df_free(df); df_free(tmp); df_free(orig);

    /* str -> float: age UINT -> STRING -> FLOAT */

    tmp = df_to_type(orig, "age", STRING);
    df = df_to_type(tmp, "age", FLOAT);

    check_float("str->float: Bob=30.0", df, 1, 1, 30.0f, 0.01f);
    df_free(df); df_free(tmp); df_free(orig);

    /* str -> bool: age UINT -> STRING -> BOOL */
    orig = df_read_csv("test_csv/basic.csv", ",");
    tmp = df_to_type(orig, "age", STRING);
    df = df_to_type(tmp, "age", BOOL);
    check_not_null("str->bool: non-NULL", df);
